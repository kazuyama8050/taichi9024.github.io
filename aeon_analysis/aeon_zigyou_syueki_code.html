            import pandas as pd
            ##グラフ使用時に日本語を使用可能
            import japanize_matplotlib 
            import matplotlib.pyplot as plt
            
            
            
            
            
            ##変数の初期化
            data_year=[]
            columns_1=[]
            columns_2=[]
            data_list_p=[]
            
            data=pd.read_csv('aeon_zigyou_profit.csv',encoding="sjis",header=0)
            print(data)
            
            ##必要な列を変数でリスト化
            data_year=data["年"]
            ##変更点1
            columns_1=data["サービス・専門店"]
            columns_2=data["営業利益"]
            
            
            ##計算
            for i in range(len(data_year)):
                x=columns_1[i]/columns_2[i]
                x=x*100
                data_list_p.append(x)
            #print(data_list_p)
            
            ##グラフの作成
            fig=plt.figure()
            plt.plot(data_year,data_list_p)
            plt.xticks(data_year)
            plt.xlabel("Year",fontsize=15)
            plt.ylabel("%",fontsize=15)
            plt.tick_params(labelsize=15)
            ##変更点2
            plt.title("サービス・専門店営業利益率",fontsize=18)
            plt.show()
